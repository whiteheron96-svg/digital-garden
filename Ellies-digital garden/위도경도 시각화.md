#folium

```
import folium
import pandas as pd
import locale

# 현재 로케일 설정 (한국)
locale.setlocale(locale.LC_ALL, 'ko_KR')

# Sample data (replace with your actual data)
data = pd.read_csv('/Users/yeseon/Documents/Python/Dangjin.csv')

# Create DataFrame
df = pd.DataFrame(data)

# Create map
m = folium.Map(location=[df['Latitude'].mean(), df['Longitude'].mean()], zoom_start=12)

# Add markers with store name and sales data in the popup
for index, row in df.iterrows():
    sales = row['Sales']
    if isinstance(sales, str):
        sales = locale.atoi(sales)  # 문자열을 숫자로 변환

    popup_content = f"<b>{row['Store Name']}</b><br>Sales: {sales:,} 백만원"  # Format sales with commas
    folium.Marker(
        location=[row['Latitude'], row['Longitude']],
        popup=popup_content,
        tooltip=row['Store Name']  # Show store name on hover
    ).add_to(m)

# Save map
m.save("dangjin_local_food_stores_with_sales.html")
```
