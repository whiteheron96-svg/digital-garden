def calc_engagement(likes, comments, reach):
    """

#### 수요일 6/10 — 클래스 기초 `50분`

**학습 (20분)**

- 클래스 선언, `__init__`, `self`
- 인스턴스 메서드

**실습 (30분)**

```python
class InstagramPost:
    def __init__(self, date, post_type, likes, comments, reach):
        self.date = date
        self.type = post_type
        self.likes = likes
        self.comments = comments
        self.reach = reach

    def engagement_rate(self):
        if self.reach == 0:
            return 0
        return (self.likes + self.comments) / self.reach * 100

    def grade(self):
        rate = self.engagement_rate()
        if rate >= 5: return "A"
        elif rate >= 3: return "B"
        else: return "C"

    def __repr__(self):
        return f"[{self.date}] {self.type} — 참여율 {self.engagement_rate():.1f}% ({self.grade()}등급)"

post1 = InstagramPost("2026-06-01", "릴스", 180, 22, 1200)
post2 = InstagramPost("2026-06-02", "이미지", 75, 8, 680)

print(post1)
print(post2)
```

- 주요 패턴: `\d`, `\w`, `\s`, `.`, `*`, `+`, `?`

**실습 (35분)**

```python
import re

# 해시태그 추출
caption = "오늘의 신제품 출시 #마케팅 #브랜드 #인스타그램 #신제품 #런칭"

print("해시태그:", hashtags)
print("해시태그 수:", len(hashtags))

# 날짜 형식 검증
dates = ["2026-06-01", "2026/06/01", "20260601", "2026-13-01"]
pattern = r"^\d{4}-\d{2}-\d{2}$"
for d in dates:
    valid = bool(re.match(pattern, d))
    print(f"{d}: {'유효' if valid else '무효'}")
```

#### 목요일 6/18 — 종합 복습 실습 `50분`

**실습 전체 (50분)** — 지금까지 배운 것 통합

```python
import csv, json
from datetime import datetime

def load_posts(filename):
    posts = []
    with open(filename, "r", encoding="utf-8-sig") as f:
        for row in csv.DictReader(f):
            posts.append({
                "date": row["날짜"],
                "type": row["유형"],
                "likes": int(row["좋아요"]),
                "comments": int(row["댓글"]),
                "reach": int(row["도달"]),
                "engagement": (int(row["좋아요"]) + int(row["댓글"])) / int(row["도달"]) * 100
            })
    return posts

def summarize(posts):
    by_type = {}
    for p in posts:
        t = p["type"]
        if t not in by_type:
            by_type[t] = []
        by_type[t].append(p["engagement"])
    return {t: sum(v)/len(v) for t, v in by_type.items()}

posts = load_posts("instagram_posts.csv")
summary = summarize(posts)

print("콘텐츠 유형별 평균 참여율:")
for type_, avg in sorted(summary.items(), key=lambda x: -x[1]):
    print(f"  {type_:10s}: {avg:.2f}%")
```

#### 금요일 6/19 — 미니 프로젝트 시작 `50분`

**목표**: 인스타그램 샘플 데이터로 간단한 분석 스크립트 완성

```python
# instagram_analyzer.py
# 한 달치 게시물 데이터 분석 → 리포트 출력

import csv
from statistics import mean, stdev

def analyze(filename):
    posts = []
    with open(filename, "r", encoding="utf-8-sig") as f:
        for row in csv.DictReader(f):
            posts.append(row)

    engagements = [
        (int(p["좋아요"]) + int(p["댓글"])) / int(p["도달"]) * 100
        for p in posts
    ]

    print("=" * 40)
    print("  인스타그램 월간 성과 리포트")
    print("=" * 40)
    print(f"총 게시물 수: {len(posts)}개")
    print(f"평균 참여율: {mean(engagements):.2f}%")
    print(f"참여율 편차: {stdev(engagements):.2f}%")
    print(f"최고 참여율: {max(engagements):.2f}%")
    print(f"최저 참여율: {min(engagements):.2f}%")

analyze("instagram_posts.csv")
```
